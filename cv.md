function pigIt(str){
  let newStr = str.split(/\s/);
    let newArr = [];
    for (let i in newStr) {
        let split = newStr[i].split("");
        if (split[0].match(/[a-z]/i)){   
            split.push(split[0]);
            split.shift();
            split.push('ay');
        }
        newArr.push(split.join(''));
    }
}
